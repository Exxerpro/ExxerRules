### Async Programming
```csharp
// Proper async patterns
public async Task<Result<T>> ProcessDataAsync<T>(
    Func<Task<T>> operation, 
    CancellationToken cancellationToken = default)
{
    try
    {
        var result = await operation().ConfigureAwait(false);
        return Result<T>.Success(result);
    }
    catch (OperationCanceledException)
    {
        return Result<T>.WithFailure("Operation was cancelled");
    }
    catch (Exception ex)
    {
        _logger.LogError(ex, "Operation failed");
        return Result<T>.WithFailure("Operation failed");
    }
}
```

## SOLID Principles

### Single Responsibility Principle
```csharp
// Each class has one reason to change
public class UserService
{
    private readonly IUserRepository _userRepository;
    private readonly IEmailService _emailService;

public class UserRepository : IUserRepository
```

### Liskov Substitution Principle
```csharp
// Derived classes can replace base classes
public abstract class Shape
{
    public abstract double Area { get; }
    public abstract double Perimeter { get; }
}

public class Rectangle : Shape
{
    public double Width { get; }
    public double Height { get; }
    
    public Rectangle(double width, double height)
    {
        Width = width;
        Height = height;
    }
    
    public override double Area => Width * Height;
    public override double Perimeter => 2 * (Width + Height);
}

public class Circle : Shape
{
    public double Radius { get; }
    
    public Circle(double radius)
    {
        Radius = radius;
    }
    
    public override double Area => Math.PI * Radius * Radius;
    public override double Perimeter => 2 * Math.PI * Radius;
}
```

## Error Handling Patterns

### Result Pattern
```csharp
public class Result<T>
{
    public bool IsSuccess { get; }
    public bool IsFailure => !IsSuccess;
    public T Value { get; }
    public string Error { get; }
    
    private Result(bool isSuccess, T value, string error)
    {
        IsSuccess = isSuccess;
        Value = value;
        Error = error;
    }
    
    public static Result<T> Success(T value) => new(true, value, null);
    public static Result<T> WithFailure(string error) => new(false, default, error);
    
    public static implicit operator Result<T>(T value) => Success(value);
}

// Usage
public async Task<Result<User>> GetUserAsync(int id)
{
    try
    {
        var user = await _userRepository.GetByIdAsync(id);
        if (user == null)
            return Result<User>.WithFailure("User not found");
        
        return Result<User>.Success(user);
    }
    catch (Exception ex)
    {
        _logger.LogError(ex, "Error retrieving user {UserId}", id);
        return Result<User>.WithFailure("An error occurred while retrieving the user");
    }
}
```

```csharp
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Unexpected error creating user");
            return Result<User>.WithFailure("An unexpected error occurred");
        }
    }
}
```

## Performance Optimization

### Memory Management
```csharp
// Use Span<T> for zero-allocation operations
public static int CountOccurrences(ReadOnlySpan<char> text, char character)
{
    int count = 0;
    for (int i = 0; i < text.Length; i++)
    {
        if (text[i] == character)
            count++;
    }
    return count;
}

// Use ArrayPool<T> for temporary arrays
public static async Task<byte[]> ProcessLargeDataAsync(Stream input)
{
    var pool = ArrayPool<byte>.Shared;
    var buffer = pool.Rent(8192);
    try
    {
        var result = new List<byte>();
        int bytesRead;
        
        while ((bytesRead = await input.ReadAsync(buffer, 0, buffer.Length)) > 0)
        {
            for (int i = 0; i < bytesRead; i++)
            {
                result.Add(buffer[i]);
            }
        }
        
        return result.ToArray();
    }
    finally
    {
        pool.Return(buffer);
    }
}
```

### Caching Strategies
```csharp
public class CachedUserService
{
    private readonly IMemoryCache _cache;
    private readonly IUserRepository _userRepository;
    private readonly TimeSpan _cacheExpiry = TimeSpan.FromMinutes(5);
    
    public async Task<User?> GetUserAsync(int id)
    {
        var cacheKey = $"user_{id}";
        
        if (_cache.TryGetValue(cacheKey, out User? cachedUser))
            return cachedUser;
        
        var user = await _userRepository.GetByIdAsync(id);
        if (user != null)
        {
            _cache.Set(cacheKey, user, _cacheExpiry);
        }
        
        return user;
    }
}
```

## Security Best Practices

### Input Validation
```csharp
public class UserValidator
{
    public Result ValidateCreateUserRequest(CreateUserRequest request)
    {
        if (request == null)
            return Result.WithFailure("Request cannot be null");
        
        if (string.IsNullOrWhiteSpace(request.FirstName))
            return Result.WithFailure("First name is required");
        
        if (string.IsNullOrWhiteSpace(request.LastName))
            return Result.WithFailure("Last name is required");
        
        if (string.IsNullOrWhiteSpace(request.Email))
            return Result.WithFailure("Email is required");
        
        if (!IsValidEmail(request.Email))
            return Result.WithFailure("Invalid email format");
        
        return Result.Success();
    }
    
    private static bool IsValidEmail(string email)
    {
        try
        {
            var addr = new System.Net.Mail.MailAddress(email);
            return addr.Address == email;
        }
        catch
        {
            return false;
        }
    }
}
```

### Authentication & Authorization

```csharp
[Authorize]
[ApiController]
[Route("api/[controller]")]
public class UsersController : ControllerBase
{
    [HttpGet("{id}")]
    [Authorize(Roles = "Admin,User")]
    public async Task<ActionResult<UserDto>> GetUser(int id)
    {
        var user = await _userService.GetUserAsync(id);

            return NotFound();
        
        return Ok(user);
    }
    
    [HttpPost]
    [Authorize(Roles = "Admin")]
    public async Task<ActionResult<UserDto>> CreateUser(CreateUserRequest request)
    {
        var result = await _userService.CreateUserAsync(request);

        return CreatedAtAction(nameof(GetUser), new { id = result.Value.Id }, result.Value);
    }
}
```

## Best Practices Summary

### Code Quality
1. **Use meaningful names** that clearly express intent
2. **Keep functions small** and focused on single responsibility
3. **Follow SOLID principles** for maintainable code
4. **Write self-documenting code** with clear structure
5. **Eliminate code duplication** through proper abstraction

### Error Handling
1. **Use Result pattern** for expected failures
2. **Handle exceptions appropriately** with proper logging
3. **Validate inputs** at system boundaries
4. **Provide meaningful error messages** to users
5. **Log errors with context** for debugging

### Performance
1. **Profile before optimizing** to identify bottlenecks
2. **Use appropriate data structures** for the use case
3. **Minimize allocations** in hot paths
4. **Implement caching** for expensive operations
5. **Use async/await** for I/O operations

### Security
1. **Validate all inputs** at system boundaries
2. **Use parameterized queries** to prevent SQL injection
3. **Implement proper authentication** and authorization
4. **Encrypt sensitive data** in transit and at rest
5. **Follow principle of least privilege**

### Testing
1. **Write tests first** using TDD approach
2. **Test all code paths** including edge cases
3. **Use meaningful test names** that describe the scenario
4. **Keep tests independent** and repeatable
5. **Maintain high test coverage** for critical paths
